#include <bits/stdc++.h>
using namespace std;
#define ll long long

void solve()
{
    int n, m;
    cin >> n >> m;
    if (n > m)
    {
        cout << "No" << '\n';
        return;
    }

    if (n % 2)
    {
        cout << "Yes" << '\n';
        for (int i = 1; i <= n - 1; i++)
            cout << 1 << " ";
        cout << m - (n - 1) << '\n';
    }
    else
    {
        if (m % 2)
        {
            cout << "No" << '\n';
            return;
        }
        cout << "Yes" << '\n';
        for (int i = 1; i <= n - 2; i++)
            cout << 1 << " ";
        cout << (m - (n - 2)) / 2 << " " << (m - (n - 2)) / 2 << '\n';
    }
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    int t;
    cin >> t;
    while (t--)
    {
        solve();
    }

    return 0;
}
