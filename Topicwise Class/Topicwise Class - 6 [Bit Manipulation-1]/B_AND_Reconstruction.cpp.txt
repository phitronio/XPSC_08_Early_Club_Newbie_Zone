#include <bits/stdc++.h>
using namespace std;
#define ll long long

void solve(int tt)
{
    int n;
    cin >> n;
    vector<int> a(n), b(n - 1);
    for(auto &u: b)
        cin >> u;

    a[0] = b[0];
    a[n - 1] = b[n - 2];
    for(int i = 1; i < n - 1; i++)
        a[i] = (b[i] | b[i - 1]);

    for(int i = 0; i < n - 1; i++)
    {
        if(b[i] != (a[i] & a[i + 1]))
        {
            cout << -1 << '\n';
            return;
        }
    }

    for(auto u: a)
        cout << u << " ";
    cout << '\n';
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    int tc = 1;
    cin >> tc;
    for (int tt = 1; tt <= tc; tt++)
    {
        solve(tt);
    }

    return 0;
}
